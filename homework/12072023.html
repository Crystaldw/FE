1) Напишите функцию, которая выводит все числа от 1 до 100. Для чисел, которые кратны 3, вместо числа выведите "Fizz", а для чисел, кратных 5, выведите "Buzz". Если число кратно и 3, и 5, выведите "FizzBuzz".

function printNumbersFizzBuzz(){

for (let i = 1; i <= 100; i ++){

if (i % 3 == 0 && i % 5 == 0){

console.log("FizzBuzz");

}else if (i % 3 == 0){

console.log("Fizz");

}else if (i % 5 == 0){

console.log("Buzz");

}else{

console.log(i);

}

}

}

printNumbersFizzBuzz();

5)Напишите функцию, которая находит наибольшее число в заданном массиве чисел

function largestNumber(array) {

let max = array[0];

for (let i = 1; i < array.length; i++) {

if (array[i] > max) {

max = array[i];

}

}

return max;

}

let numbers = [10, 6, 8, 32, 22, 7];

let largNumber = largestNumber(numbers);

console.log("Наибольшее число из массива:", largNumber);
